box-sizing Property

/**
 * BOX-SIZING
 * 
 * Controls how width and height are calculated.
 */

/* content-box (DEFAULT) */
.element {
    box-sizing: content-box;
    width: 200px;
    padding: 20px;
    border: 5px solid black;
}
/* Total width = 200 + 20 + 20 + 5 + 5 = 250px */

/* border-box (RECOMMENDED) */
.element {
    box-sizing: border-box;
    width: 200px;
    padding: 20px;
    border: 5px solid black;
}
/* Total width = 200px (padding and border included) */
/* Content width = 200 - 20 - 20 - 5 - 5 = 150px */

/* Best practice: Apply to all elements */
*, *::before, *::after {
    box-sizing: border-box;
}

Margin

/**
 * MARGIN - Space OUTSIDE the element
 */

.element {
    margin: 10px;                    /* All sides */
    margin: 10px 20px;               /* Vertical | Horizontal */
    margin: 10px 20px 30px;          /* Top | Horizontal | Bottom */
    margin: 10px 20px 30px 40px;     /* Top | Right | Bottom | Left */
    
    /* Individual sides */
    margin-top: 10px;
    margin-right: 20px;
    margin-bottom: 30px;
    margin-left: 40px;
    
    /* Auto margins for centering */
    margin: 0 auto;  /* Center horizontally */
    
    /* Negative margins */
    margin-top: -10px;  /* Pull element up */
}

/**
 * MARGIN COLLAPSE
 * 
 * Vertical margins collapse (take the larger value).
 * Horizontal margins do NOT collapse.
 */

/* Example: */
.box1 { margin-bottom: 20px; }
.box2 { margin-top: 30px; }
/* Space between = 30px (not 50px!) */

/* Margin collapse does NOT happen with: */
/* - Flexbox items */
/* - Grid items */
/* - Floated elements */
/* - Absolutely positioned elements */
/* - Elements with overflow other than visible */

Padding

/**
 * PADDING - Space INSIDE the element (between content and border)
 */

.element {
    padding: 10px;                   /* All sides */
    padding: 10px 20px;              /* Vertical | Horizontal */
    padding: 10px 20px 30px;         /* Top | Horizontal | Bottom */
    padding: 10px 20px 30px 40px;    /* Top | Right | Bottom | Left */
    
    /* Individual sides */
    padding-top: 10px;
    padding-right: 20px;
    padding-bottom: 30px;
    padding-left: 40px;
}

/* Padding CANNOT be negative */
/* Padding is included in background area */
/* Padding does NOT collapse */

Border

/**
 * BORDER - Between padding and margin
 */

.element {
    /* Shorthand */
    border: 1px solid black;
    
    /* Individual properties */
    border-width: 1px;
    border-style: solid;  /* solid | dashed | dotted | double | none */
    border-color: black;
    
    /* Individual sides */
    border-top: 2px dashed red;
    border-right: 3px solid blue;
    border-bottom: 4px dotted green;
    border-left: 5px double orange;
    
    /* Border radius */
    border-radius: 10px;              /* All corners */
    border-radius: 10px 20px;         /* TL+BR | TR+BL */
    border-radius: 10px 20px 30px 40px; /* TL | TR | BR | BL */
    border-radius: 50%;               /* Circle (if square) */
}

Width and Height

/**
 * WIDTH AND HEIGHT
 */

.element {
    /* Fixed size */
    width: 200px;
    height: 100px;
    
    /* Percentage (relative to parent) */
    width: 50%;
    height: 100%;  /* Needs parent height defined */
    
    /* Viewport units */
    width: 100vw;   /* 100% of viewport width */
    height: 100vh;  /* 100% of viewport height */
    
    /* Min/Max constraints */
    min-width: 200px;
    max-width: 800px;
    min-height: 100px;
    max-height: 500px;
    
    /* Auto (default) */
    width: auto;   /* Shrink to content or expand to fill */
    height: auto;  /* Shrink to content */
    
    /* fit-content */
    width: fit-content;  /* Shrink to content, max at container */
    
    /* min-content / max-content */
    width: min-content;  /* Smallest possible width */
    width: max-content;  /* No wrapping, natural width */
}

Display Property

/**
 * DISPLAY - How element participates in layout
 */

/* Block: Takes full width, starts on new line */
display: block;
/* Examples: div, p, h1-h6, section, article */

/* Inline: Only takes content width, no width/height */
display: inline;
/* Examples: span, a, strong, em */

/* Inline-block: Inline but respects width/height */
display: inline-block;

/* None: Removes from layout completely */
display: none;

/* Flex: Flexbox container */
display: flex;

/* Grid: Grid container */
display: grid;

/* Inline variants */
display: inline-flex;
display: inline-grid;

Overflow

/**
 * OVERFLOW - What happens when content exceeds container
 */

.element {
    overflow: visible;  /* Default - content overflows */
    overflow: hidden;   /* Clip content */
    overflow: scroll;   /* Always show scrollbars */
    overflow: auto;     /* Scrollbars only when needed */
    
    /* Individual axes */
    overflow-x: hidden;
    overflow-y: auto;
    
    /* Text overflow */
    white-space: nowrap;
    overflow: hidden;
    text-overflow: ellipsis;  /* Shows "..." */
}

  • What is box-sizing: border-box? Width/height includes padding and border
  • What is margin collapse? Adjacent vertical margins combine (take larger value)
  • Difference between margin and padding? Margin is outside, padding is inside the border
